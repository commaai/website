{{- /* Replicates the SvelteKit optimize-image preprocessor output: .src logical image path, e.g. /img/products/comma-four/four_back.png .attrs map of extra attributes (alt, loading, class, style, width, ...) Variants exist as siblings generated by scripts/optimize-images.mjs. jpg/jpeg/png/gif sources get treatment; everything else renders a bare . */ -}} {{- $src := .src -}} {{- $ext := path.Ext $src -}} {{- $stem := strings.TrimSuffix $ext $src -}} {{- $attrs := .attrs | default dict -}} {{- if in (slice ".jpg" ".jpeg" ".png" ".gif") (lower $ext) -}} {{- else -}} {{- end -}}
